weui.js custom

修改weui.js 源码，添加多种提醒样式

代码演示

weui.topTips('请填写正确的字段');
weui.topTips('请填写正确的字段', 3000);
weui.topTips('请填写正确的字段', function(){ console.log('close') });
weui.topTips('请填写正确的字段', {
    duration: 3000,
    className: 'custom-classname',
    callback: function(){ console.log('close') }
});

// 主动关闭
var $topTips = weui.topTips('请填写正确的字段');
$topTips.hide(function() {
     console.log('`topTips` has been hidden');
});
weui.topTips('成功', 'success');
weui.topTips('失败', {
     duration: 3000,
     className: 'custom-classname',
     style: 'error',
     callback: function(){ console.log('close') }
});
weui.topTips('警告', 'warning');

toast API

参数 说明 类型 默认值
content 报错的文字 string -
options 多少毫秒后消失 number | function | object -
[options.duration] 多少毫秒后消失 number 3000
[options.callback] 消失后的回调 function -
[options.className] 自定义类名 string -
[options.style] 类型, success | warning | error string error
