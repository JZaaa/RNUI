weui.js custom

gallery 图片浏览,新增删除按钮隐藏

代码演示

function actionSheet() {
    let url = './doc/img/pic_article.png';
    let gallery = weui.gallery(url, {
        className: 'custom-classname',
        delete: false, //默认为true，隐藏删除按钮
        onDelete: function(){
            weui.confirm('确定删除该图片？',
                function(){
                    console.log('删除');
                    hide();
                }
            );
        }
    });

    function hide() {
        gallery.hide(function() {
            console.log('`gallery` has been hidden');
        });
    }
}
